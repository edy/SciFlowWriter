\usepackage{amssymb}
\setcounter{tocdepth}{3}
\usepackage{graphicx}
\usepackage{url}
\usepackage{blindtext}
\usepackage[english]{babel}
\usepackage{todonotes}
\usepackage{booktabs}
\usepackage{subfig}
\usepackage[utf8]{inputenc}
\usepackage{ulem}

\makeatletter
\define@key{Gin}{Trim}
           {\let\Gin@viewport@code\Gin@trim\expandafter\Gread@parse@vp#1 \\}
\makeatother

\newtoks\TrimVal


\begin{document}

\mainmatter  % start of an individual contribution

% first the title is needed
\title{<%= metadata.title %>}

\subtitle{<%= metadata.subtitle %>}
% a short form should be given in case it is too long for the running head
%\titlerunning{Awareness}

% the name(s) of the author(s) follow(s) next
%
% NB: Chinese authors should write their first names(s) in front of
% their surnames. This ensures that the names appear correctly in
% the running heads and the author index.
%
%\author{Wolfgang Reinhardt \and Christian Mletzko}
<%
var authors = [];
for (author in users) {
	authors.push(users[author].name);
}
%>
\author{<%= authors.join(' \\and ') %>}

%\institute{University of Paderborn\\Department of Computer Science, Computer Science Education Group\\ Fuerstenallee 11, 33102 Paderborn, Germany\\                        
%\email{\{wolle,letris\}@upb.de}}

<%
var emails = [];
for (author in users) {
	emails.push(users[author].email);
}
%>

% this is a template for lecture notes, so we use only the address from
% the first user. it's the owner of the pad
\institute{<%= users[0].institute1 %>, <%= users[0].institute2 %>\\ <%= users[0].institute3 %>\\
<%= users[0].street %>, <%= users[0].city %>, <%= users[0].country %>\\
\email{<%= emails.join(', ') %>}}

\maketitle

\begin{abstract}
<%= metadata.abstract %>

\keywords{<%= metadata.keywords %>}
\end{abstract}


<%= latexExport %>


\nocite{*}
\bibliographystyle{plain}
\bibliography{pad}


\end{document}
